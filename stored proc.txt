DECLARE
    @ReviewID nvarchar(50)
   ,@DateActiondue NVARCHAR(200)
   ,@Active NVARCHAR(100)
   ,@ActionNeeded NVARCHAR(200)
  

--AS 
--    SET NOCOUNT ON;
    SELECT  RR.RevID
	      , RR.Project_Description
		  , RA.Date_Actions_Due
		  ,RR.Active
		  , RA.Actions_Needed
		  ,RR.Project_Latitude
		  ,RR.Project_Longitude
    FROM Review.Review RR
	INNER JOIN Review.REV_Action RA ON RA.RevID = RR.RevID
    WHERE (RA.Date_Actions_Due = ISNULL(@DateActiondue, RA.Date_Actions_Due)) 
  AND (RR.Active =ISNULL(@Active, RR.Active))
  AND (RA.Actions_Needed = ISNULL(@ActionNeeded, RA.Actions_Needed))
  AND RR.REVID = 1 ;
