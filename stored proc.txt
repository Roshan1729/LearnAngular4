ALTER PROCEDURE [dbo].[sp_GetSearchResults] 
    @ReviewID nvarchar(50)
AS 
    SET NOCOUNT ON;
    SELECT RevID
	      , RR.Project_Description
		  , RA.Date_Action_Due
		  ,RR.Active
		  , RA.Actions_Needed
		  ,RR.Project_Latitude
		  ,RR.Project_Longitude
    FROM Review.Review RR
	INNER JOIN [Review.rev_action] RA ON RA.[RevID] = RR.[RevID]
    WHERE RevID = @ReviewID;
