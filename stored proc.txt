ALTER PROCEDURE [dbo].[sp_GetSearchResults] 
    @ReviewID nvarchar(50)
   --,@DateActiondue NVARCHAR(200)
   ,@Active NVARCHAR(100)
   ,@ActionNeeded NVARCHAR(200)
  

AS 
    SET NOCOUNT ON;
    SELECT  RR.RevID
	      , RR.Project_Description
		  , RA.Date_Actions_Due
		  ,RR.Active
		  , RA.Actions_Needed
		  ,RR.Project_Latitude
		  ,RR.Project_Longitude
    FROM Review.Review RR
	INNER JOIN Review.REV_Action RA ON RA.RevID = RR.RevID
    WHERE (RR.REVID = @ReviewID)
	AND (@Active IS NULL OR RR.Active = @Active)
  
  AND (@ActionNeeded IS NULL OR RA.ActionNeeded = @ActionNeeded) ;
