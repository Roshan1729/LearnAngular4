ALTER PROCEDURE [dbo].[sp_SearchFields]   
AS   
Begin
    SET NOCOUNT ON;  
    Declare @listCityName varchar(MAX)
	select DISTINCT name as CityName
		FROM  REVIEW.LK_CITY c;

		Declare @listBoroughName varchar(MAX)
	select DISTINCT name as BoroughName
		FROM  REVIEW.LK_BOROUGH b;

		Declare @listCountyName varchar(MAX)
	select DISTINCT name as CountyName
		FROM  REVIEW.LK_COUNTY co;

		Declare @listPlaceName varchar(MAX)
	select DISTINCT name as PlaceName
		FROM  REVIEW.LK_PLACE p;

		Declare @listTownName varchar(MAX)
	select DISTINCT name as TownName
		FROM  REVIEW.LK_TOWN t;

		Declare @listVillageName varchar(MAX)
	select DISTINCT name as VillageName
		FROM  REVIEW.LK_VILLAGE v;

		Declare @listWaterBodyName varchar(MAX)
	select DISTINCT name as WaterBodyName
		FROM  REVIEW.LK_WATERBODY wb;

		SELECT @listCityName AS LocationName
		,@listBoroughName AS BoroughName
		,@listCountyName AS CountyName
		,@listPlaceName AS PlaceName
		,@listTownName AS TownName
		,@listVillageName AS VillageName
		,@listWaterBodyName AS WaterBodyName
		

	End;
