ALTER PROCEDURE [dbo].[sp_GetSearchResults] 
    @ReviewID int
	,@RevIDYear numeric
	,@ActionCD varchar(50)
	,@AgencyAppNum varchar(255)
	,@ConsultantProjNum varchar(50)
   ,@ProjectDescription varchar(2000)
   ,@ProjectTitle varchar(255)
   --,@DateActiondue DATETIME2(0)
   ,@Active VARCHAR(1)
   --,@ActionNeeded VARCHAR(2000)
   --,@Latitude float
   --,@longitude float
  

AS 
    SET NOCOUNT ON;
    SELECT  RR.RevID
	      , RR.Project_Description
		  , RA.Date_Actions_Due
		  ,RR.Active
		  , RA.Actions_Needed
		  ,RR.Project_Latitude
		  ,RR.Project_Longitude
    FROM Review.Review RR
	INNER JOIN Review.REV_Action RA ON RA.RevID = RR.RevID
    WHERE (RR.REVID = @ReviewID)
	--AND (@Active IS NULL OR RR.Active = @Active)
  
  --AND (@ActionNeeded IS NULL OR RA.Actions_Needed = @ActionNeeded) 
  --AND (@Latitude IS NULL OR RR.Project_Latitude = @Latitude)
  --AND (@longitude IS NULL OR RR.Project_Longitude = @longitude);
