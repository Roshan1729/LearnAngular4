ALTER PROCEDURE [dbo].[ModifyUserRole] 
   	@Login varchar (100)
    ,@NewRoleType varchar (100)
	,@ExistingRole varchar output
	
AS 
    SET NOCOUNT ON;
	select  @ExistingRole = [roletype]  from [SVC\_rreshma].[users] where UserName = @Login;

    Update [SVC\_rreshma].[users]  set 
	roletype = @NewRoleType  where
	UserName = @Login;
