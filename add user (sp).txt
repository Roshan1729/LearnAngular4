ALTER PROCEDURE [dbo].[AddUser] 
    @FirstName varchar(50)
	,@Lastname varchar(20)
	,@Login nvarchar (100)
	,@Password Nvarchar(255)
	,@RoleType varchar(20)
	,@CreatedDate Datetime
	,@DeactivatedDate Datetime

	AS
BEGIN
    SET NOCOUNT ON;
	IF EXISTS(SELECT * FROM [SVC\_rreshma].[users] WHERE UserName = @Login)
    BEGIN
        SELECT 'FALSE'
    END
    ELSE
    BEGIN

   
    Insert into [SVC\_rreshma].[users] values ( @Login, 
	@Password,@FirstName, @Lastname, @RoleType, @CreatedDate, @DeactivatedDate)
	END
	END
	GO
